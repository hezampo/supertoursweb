function getItemChecked(parametro){

    var radios = document.getElementsByName(parametro);
    var valor = -1;

    for (var x=0; x < radios.length; x++) {
        if (radios[x].checked) {
            valor =  radios[x].value;
            break;
        }
    }

    return valor;
}
